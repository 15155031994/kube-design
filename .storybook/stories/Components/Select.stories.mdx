### Custom Select Render

<Preview>
	<Story name="select-render">
		{() => {
			const options = [
				{ value: "CentOS", label: "CentOS 5.8 32bit" },
				{ value: "Debian", label: "Debian Jessie 8.1 64bit" },
				{ value: "Ubuntu", label: "Ubuntu Server 14.04.3 LTS 64bit", disabled: true },
				{ value: "Windows", label: "Windows Server 2003 R2" },
			];
			const optionRenderer = (option) => (
				<span
					className="option-with-icon"
					style={{ display: "flex", alignItems: "center" }}
				>
					<Icon
						name="kubesphere"
						size="small"
						style={{ marginRight: 10 }}
						type="dark"
					/>
					{option.label}
				</span>
			);
			const valueRenderer = (option) => (
				<span
					className="option-with-icon"
					style={{ display: "flex", alignItems: "center" }}
				>
					<Icon
						name="kubesphere"
						size="small"
						type="coloured"
						style={{ marginRight: 10 }}
					/>
					<span style={{ color: "#00aa72" }}>{option}</span>
				</span>
			);
			return (
				<Select
					clearable
					name="select-render"
					optionRenderer={optionRenderer}
					valueRenderer={valueRenderer}
					options={options}
				/>
			);
		}}
	</Story>
</Preview>
